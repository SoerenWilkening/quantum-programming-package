**1. Add _tree_size() method:**

```python
def _tree_size(self):
    """Compute total number of nodes in the tree.

    For uniform branching d at each level with max_depth n:
    T = 1 + d + d^2 + ... + d^n = (d^(n+1) - 1) / (d - 1)

    For variable branching, uses the maximum branching degree
    at each level (conservative upper bound for iteration count).

    Returns
    -------
    int
        Total node count.
    """
    total = 1  # root
    level_count = 1
    for level_idx in range(self.max_depth):
        level_count *= self.branching[level_idx]
        total += level_count
    return total
```

**2. Add _measure_root_overlap() method:**

This method constructs a fresh circuit, builds a new QWalkTree with the same parameters, applies walk_step num_steps times, extracts the statevector, and computes root state overlap probability.

```python
def _measure_root_overlap(self, num_steps):
    """Build a fresh circuit with num_steps walk steps and measure root overlap.

    Constructs a new circuit and QWalkTree with identical parameters,
    applies walk_step the specified number of times, then computes
    |<r|psi>|^2 where |r> is the root state.

    Parameters
    ----------
    num_steps : int
        Number of walk steps to apply.

    Returns
    -------
    float
        Probability of measuring the root state (|<r|psi>|^2).
    """
    import quantum_language as ql
    import qiskit.qasm3
    from qiskit import transpile
    from qiskit_aer import AerSimulator

    # Save no reference to outer circuit -- ql.circuit() replaces global state
    ql.circuit()
    inner_tree = QWalkTree(
        max_depth=self.max_depth,
        branching=self.branching[0] if len(set(self.branching)) == 1 else list(self.branching),
        predicate=self._predicate,
        max_qubits=self.max_qubits,
    )

    for _ in range(num_steps):
        inner_tree.walk_step()

    qasm_str = ql.to_openqasm()

    # Statevector simulation
    circuit = qiskit.qasm3.loads(qasm_str)
    circuit.save_statevector()
    sim = AerSimulator(method="statevector", max_parallel_threads=4)
    result = sim.run(transpile(circuit, sim)).result()
    sv = np.asarray(result.get_statevector())

    # Root state index: h[max_depth]=|1>, all others |0>
    # Qiskit little-endian: qubit k set = bit k of statevector index
    h_root = inner_tree._height_qubit(inner_tree.max_depth)
    root_idx = 1 << h_root
    return float(abs(sv[root_idx]) ** 2)
```

**3. Add detect() method:**

```python
def detect(self, max_iterations=None):
    """Detect whether a marked vertex (solution) exists in the backtracking tree.

    Implements Montanaro's Algorithm 1: iterative power-method detection.
    Applies the walk step U raised to increasing powers (1, 2, 4, 8, ...),
    measuring root state overlap after each. If overlap probability drops
    below 3/8 at any power level, a solution exists.

    Each power level is a separate circuit execution. The method does not
    modify the caller's circuit state.

    Parameters
    ----------
    max_iterations : int, optional
        Maximum number of walk steps in the largest power level.
        Auto-computed from tree size as ceil(sqrt(T / n)) if not provided,
        where T is total node count and n is max_depth.

    Returns
    -------
    bool
        True if a marked vertex (solution) is detected, False otherwise.
    """
    if max_iterations is None:
        T = self._tree_size()
        n = self.max_depth
        max_iterations = max(4, int(math.ceil(math.sqrt(T / n))))

    threshold = 3.0 / 8.0

    power = 1
    while power <= max_iterations:
        root_prob = self._measure_root_overlap(power)
        if root_prob < threshold:
            return True  # Solution detected
        power *= 2

    return False  # No solution detected
```

```python
def _simulate_statevector(qasm_str):
    """Run QASM through Qiskit Aer and return statevector as numpy array."""
    circuit = qiskit.qasm3.loads(qasm_str)
    circuit.save_statevector()
    sim = AerSimulator(method="statevector", max_parallel_threads=4)
    result = sim.run(transpile(circuit, sim)).result()
    return np.asarray(result.get_statevector())


# ---------------------------------------------------------------------------
# Test predicates for detection
# ---------------------------------------------------------------------------


def _accept_all_leaves_predicate(node):
    """Predicate that accepts ALL leaf nodes.

    At leaf level, sets is_accept=|1>. At internal nodes, neither accepts
    nor rejects. This means the tree has solutions at every leaf.
    """
    is_accept = ql.qbool()
    is_reject = ql.qbool()
    # Simple accept-all: leaves are all solutions
    # We don't set is_accept here because detection works by measuring
    # how the walk distributes amplitude, not by reading accept qubits directly.
    # For the walk to detect solutions, the predicate must mark some vertices.
    # The simplest approach: use a predicate that never rejects anything.
    return (is_accept, is_reject)


def _never_accept_predicate(node):
    """Predicate that never accepts any node (no solutions).

    Both is_accept and is_reject stay |0>.
    """
    is_accept = ql.qbool()
    is_reject = ql.qbool()
    return (is_accept, is_reject)
```
