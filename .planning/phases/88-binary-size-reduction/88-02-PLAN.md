Benchmark using this pattern (run in Python):
```python
import time
import quantum_language as ql

def benchmark_circuit_gen():
    """Benchmark representative circuit generation workload."""
    times = []
    for run in range(3):
        ql.reset_circuit()
        start = time.perf_counter()
        # Representative workload: arithmetic operations that exercise the C backend
        a = ql.qint(value=5, bits=8)
        b = ql.qint(value=3, bits=8)
        c = a + b       # Addition
        d = a - b       # Subtraction
        e = a * b       # Multiplication (heaviest operation)
        f = c + d       # Chained operations
        elapsed = time.perf_counter() - start
        times.append(elapsed)
        print(f"  Run {run+1}: {elapsed*1000:.2f} ms")
    times.sort()
    median = times[len(times) // 2]
    print(f"  Median: {median*1000:.2f} ms")
    return median
```
